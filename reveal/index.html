Local Development

10-Minute Setup

Clone & Configure


git clone https://github.com/jag18729/guard-quote.git
cd guard-quote
cp .env.example .env

Environment Variables


# .env
DB_HOST=100.66.167.62
DB_PASSWORD=[see .env]

REDIS_HOST=100.66.167.62
REDIS_PASSWORD=[see .env]

JWT_SECRET=guardquote_local_dev_secret_32chars
ML_ENGINE_SECRET=[see .env]

Start Services


# Terminal 1 - Backend
cd backend && bun install && bun run dev
# → http://localhost:3000

# Terminal 2 - Frontend
cd frontend && bun install && bun run dev
# → http://localhost:5173

# Terminal 3 - ML Engine
cd ml-engine
python -m venv .venv && source .venv/bin/activate
pip install -e . && uvicorn src.main:app --reload --port 8000
# → http://localhost:8000

Verify


# Backend health
curl http://localhost:3000/health
# {"status":"healthy","database":"connected"}

# ML Engine
curl http://localhost:8000/health
# {"status":"healthy","model_status":"active"}

# Frontend
open http://localhost:5173
